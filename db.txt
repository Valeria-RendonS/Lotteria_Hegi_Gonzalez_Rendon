create database lotteria_db;
use lotteria_db;

create table utente(
	id int primary key auto_increment,
	nome varchar(50) not null,
	cognome varchar(50) not null,
	email varchar(100) not null,
	user_name varchar(50) not null,
	n_telefono int(10) not null check(n_telefono = 10),
	data_nascita date not null,
	password varchar(300) not null,
	crediti int not null,
	data_registrazione date not null default current_date,
	last_login date not null,
	codice_OTP int (6) null,
	bloccato bool not null default false,
	ruolo varchar(20) not null
);


create table lotteria(
	id int primary key auto_increment,
	nome varchar(50) not null,
	descrizione varchar(200) not null,
	incasso int not null default 0,
	erogato int not null default 0,
	prezzo_biglietto int not null,
	n_biglietti_totali int not null,
	n_biglietti_venduti int not null set 0,
	data_inizio_acquisti date not null,
	data_estrazione date null,
	data_fine_acquisti date null,
	stato varchar(20) not null check(stato in ("attivo", "finito", "annullato")),
	biglietto_vincente int null,

	foreign key (id_utente) references utente(id)
);



create table biglietto(
	id int primary key auto_increment,
	numero_biglietto int not null,
	
	foreing key (id_lotteira) references lotteria(id) on delete cascade,
	foreing key (id_utente) references utente(id)
);

alter table lotteria foreign key (biglietto_vincente) references biglietto(id);
